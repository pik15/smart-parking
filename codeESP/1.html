#include #include #include #include #include const char* ssid = "opik"; const char* password = "11111111"; const char* mqtt_server = "test.mosquitto.org"; WiFiClient espClient; PubSubClient client(espClient); Servo palangMasuk; #define SS_PIN 5 #define RST_PIN 22 #define PIN_TRIG 12 #define PIN_ECHO 14 #define PIN_SERVO 13 MFRC522 mfrc522(SS_PIN, RST_PIN); int sisaSlotSaatIni = 10; bool mobilMenunggu = false; bool mobilSudahLewat = false; bool statusTerdeteksiDikirim = false; bool menungguValidasiDatabase = false; long bacaJarak() { digitalWrite(PIN_TRIG, LOW); delayMicroseconds(2); digitalWrite(PIN_TRIG, HIGH); delayMicroseconds(10); digitalWrite(PIN_TRIG, LOW); long durasi = pulseIn(PIN_ECHO, HIGH, 20000); long jarak = durasi * 0.034 / 2; return (jarak == 0) ? 999 : jarak; } void callback(char* topic, byte* payload, unsigned int length) { String message = ""; for (int i = 0; i < length; i++) message += (char)payload[i]; String topicStr = String(topic); if (topicStr == "opik/parking/status_slot") { sisaSlotSaatIni = message.toInt(); } else if (topicStr == "opik/parking/respon_masuk") { if (menungguValidasiDatabase) { if (message == "ACC_BUKA") { Serial.println("-> Akses Masuk Diterima! Membuka Palang..."); palangMasuk.write(0); mobilMenunggu = true; mobilSudahLewat = false; menungguValidasiDatabase = false; delay(500); } else if (message == "ACC_REJECT") { Serial.println("-> Akses Masuk Ditolak."); delay(1000); client.publish("opik/parking/masuk", "REQ_SELESAI"); menungguValidasiDatabase = false; statusTerdeteksiDikirim = false; } } } } void setup() { Serial.begin(115200); SPI.begin(); mfrc522.PCD_Init(); pinMode(PIN_TRIG, OUTPUT); pinMode(PIN_ECHO, INPUT); ESP32PWM::allocateTimer(0); palangMasuk.attach(PIN_SERVO, 500, 2400); palangMasuk.write(90); WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\nWiFi Terhubung (ESP Masuk)!"); client.setServer(mqtt_server, 1883); client.setCallback(callback); } void reconnect() { while (!client.connected()) { String clientId = "Wemos_GateIn_Opik_" + String(random(0, 9999)); if (client.connect(clientId.c_str())) { client.subscribe("opik/parking/status_slot"); client.subscribe("opik/parking/respon_masuk"); Serial.println("MQTT Masuk Terhubung ke Mosquitto!"); } else { delay(2000); } } } void loop() { if (!client.connected()) reconnect(); client.loop(); long jarak = bacaJarak(); if (!mobilMenunggu && !menungguValidasiDatabase && jarak < 15) { if (!statusTerdeteksiDikirim) { client.publish("opik/parking/masuk", "MOBIL_DEKAT"); statusTerdeteksiDikirim = true; Serial.println("-> Mobil dekat gerbang masuk, silakan scan RFID..."); } if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) { if (sisaSlotSaatIni > 0) { String uidString = ""; for (byte i = 0; i < mfrc522.uid.size; i++) { uidString += String(mfrc522.uid.uidByte[i] < 0x10 ? "0" : ""); uidString += String(mfrc522.uid.uidByte[i], HEX); } uidString.toUpperCase(); Serial.print("-> UID Masuk: "); Serial.println(uidString); client.publish("opik/parking/uid_scan", uidString.c_str()); menungguValidasiDatabase = true; } else { client.publish("opik/parking/masuk", "REQ_PENUH"); } mfrc522.PICC_HaltA(); mfrc522.PCD_StopCrypto1(); } } else if (!mobilMenunggu && !menungguValidasiDatabase && jarak >= 15 && statusTerdeteksiDikirim) { client.publish("opik/parking/masuk", "REQ_SELESAI"); statusTerdeteksiDikirim = false; } if (mobilMenunggu) { if (jarak < 15) { mobilSudahLewat = true; } if (mobilSudahLewat && jarak > 20 && jarak != 999) { delay(800); if (bacaJarak() > 20) { Serial.println("-> Mobil masuk aman. Menutup palang."); client.publish("opik/parking/masuk", "REQ_TUTUP"); palangMasuk.write(90); delay(1500); mobilMenunggu = false; mobilSudahLewat = false; statusTerdeteksiDikirim = false; client.publish("opik/parking/masuk", "REQ_SELESAI"); } } } delay(50); }
